\section{Usage}
\begin{frame}[fragile]{Style usage}
	Include using:\\
	\verb+\usetheme{CleanEUR}+ \\
	\verb+\usepackage[light,sfdefault]{roboto}+\\
	\vskip\baselineskip
	Options to the package
	\begin{itemize}
		\item titletoc: show table of contents on the titlepage
		\item nologo: don't show a logo on the titlepage and in the 
			bottom left corner
	\end{itemize}
	Example:\\
	\verb+\usetheme[nologo,titletoc]{CleanEUR}+\\
	\verb+\usepackage[light,sfdefault]{roboto}+
\end{frame}
